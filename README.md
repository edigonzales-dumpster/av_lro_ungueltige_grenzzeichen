## Query
```
INSERT INTO 
    av_lro_neu.t_grenzpunkte (geometrie)

SELECT 
    ST_SnapToGrid(grenzpunkt_alt.geometrie, 0.001) AS geometrie
FROM 
    av_lro_neu.liegenschaften_liegenschaft_geometrie AS liegenschaft_neu
    INNER JOIN av_lro_alt.liegenschaften_grenzpunkt AS grenzpunkt_alt
    ON ST_Intersects(grenzpunkt_alt.geometrie, ST_Buffer(_geom, 5))
WHERE 
    grenzpunkt_alt.punktzeichen != 'unversichert'

EXCEPT 
    
SELECT 
    ST_SnapToGrid(grenzpunkt_neu.geometrie, 0.001) AS geometrie
FROM 
    av_lro_neu.liegenschaften_grenzpunkt  AS grenzpunkt_neu
;
```
